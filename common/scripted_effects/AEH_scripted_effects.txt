AEH_return_stalemate = {
	#test if there is a frontline in enemy territory
	every_country = {
		limit = {
			AEH_is_fighting_in_alliance = yes
		}
		every_enemy_country = {
			limit = {
				AEH_has_frontline = yes
			}
			log = "[?THIS.GetName] Enemy"
			AEH_has_frontline_log = yes
		}
	}
	log = "DONE"
}

AEH_has_frontline_log = {
	every_controlled_state = {
		every_neighbor_state = {
			limit = {
				CONTROLLER = {
					has_war_with = PREV.PREV.PREV
				}
			}
			log = "[?THIS.GetName] State Frontline"
		}
	}
}

AEH_AI_set_minimim_training_time_fix = {
	if = {
		limit = {
			OR = {
				has_tech = superior_firepower
				has_tech = mobile_warfare
			}
		}
		add_ideas = AEH_AI_minimum_training_time_fix_60
	}
	if = {
		limit = {
			has_tech = trench_warfare
		}
		add_ideas = AEH_AI_minimum_training_time_fix_40
	}
	if = {
		limit = {
			has_tech = mass_assault
		}
		add_ideas = AEH_AI_minimum_training_time_fix_20
	}
}

AEH_AI_remove_minimim_training_time_fix = {
	if = {
		limit = {
			has_idea = AEH_AI_minimum_training_time_fix_60
		}
		remove_ideas = AEH_AI_minimum_training_time_fix_60
	}
	if = {
		limit = {
			has_idea = AEH_AI_minimum_training_time_fix_40
		}
		remove_ideas = AEH_AI_minimum_training_time_fix_40
	}
	if = {
		limit = {
			has_idea = AEH_AI_minimum_training_time_fix_20
		}
		remove_ideas = AEH_AI_minimum_training_time_fix_20
	}
}
