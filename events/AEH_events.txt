add_namespace = AEH
country_event = {
	id = AEH.1
	fire_only_once = yes
	is_triggered_only = yes
	hidden = yes
	immediate = {
		##### INDUSTRY BOOSTS
		log = "Industry Balance Init"
		set_temp_variable = {
			loop = 0
		}
		set_temp_variable = {
			USA_debug = 0
		}
		# log = "USA init civs number = [?USA.num_of_civilian_factories]"
		while_loop_effect = {
			limit = {
				check_variable = {
					loop < 20
				}
			}
			USA = {
				random_owned_state = {
					limit = {
						# is_fully_controlled_by = PREV
						NOT = {
							has_state_flag = AEH_industry_balance
						}
					}
					add_extra_state_shared_building_slots = 2
					add_building_construction = {
						type = industrial_complex
						level = 2
						instant_build = yes
					}
					add_to_temp_variable = {
						USA_debug = 2
					}
					# log = "State = [?THIS.GetID]"
					set_state_flag = AEH_industry_balance
				}
			}
			set_temp_variable = {
				moduloVariable = loop
			}
			modulo_temp_variable = {
				moduloVariable = 2
			}
			if = {
				limit = {
					check_variable = {
						moduloVariable = 0
					}
				}
				SOV = {
					random_owned_state = {
						limit = {
							# is_fully_controlled_by = PREV
							distance_to = {
								value < 8000
								target = 408
							}
							NOT = {
								has_state_flag = AEH_industry_balance
							}
						}
						add_extra_state_shared_building_slots = 2
						add_building_construction = {
							type = industrial_complex
							level = 2
							instant_build = yes
						}
						# log = "One added SOV"
						set_state_flag = AEH_industry_balance
					}
				}
				JAP = {
					random_owned_state = {
						limit = {
							NOT = {
								has_state_flag = AEH_industry_balance
							}
							is_core_of = PREV
						}
						add_extra_state_shared_building_slots = 1
						add_building_construction = {
							type = industrial_complex
							level = 1
							instant_build = yes
						}
						# log = "One added JAP"
						set_state_flag = AEH_industry_balance
					}
				}
			}
			add_to_temp_variable = {
				loop = 1
			}
		}
		# log = "USA final civs number = [?USA.num_of_civilian_factories]"
		# log = "USA number added : [?USA_debug]"
		log = "Industry Balance Completed"
		### Designers
	}
	option = {
	}
}
